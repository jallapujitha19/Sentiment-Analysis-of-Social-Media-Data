vfrom flask import Flask, request, jsonify from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer app = Flask(__name__) @app.route('/') def index(): return open('index.html').read() @app.route('/sentiment-analysis', methods=['POST']) def sentiment_analysis(): data = request.get_json() text = data['text'] # Perform sentiment analysis using VADER Sentiment Analysis analyzer = SentimentIntensityAnalyzer() sentiment_score = analyzer.polarity_scores(text) # Determine sentiment based on the compound score if sentiment_score['compound'] >= 0.05: sentiment = "Positive" elif sentiment_score['compound'] <= -0.05: sentiment = "Negative" else: sentiment = "Neutral" return jsonify({"sentiment": sentiment}) if __name__ == '__main__': app.run()
